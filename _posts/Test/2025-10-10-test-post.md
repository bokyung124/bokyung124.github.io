# --- Slug 변환 함수 ---
def slugify(text):
    # 한글을 제외한 모든 비-알파벳/숫자 문자를 제거
    text = re.sub(r"[^a-zA-Z0-9\u3131-\u3163\uac00-\ud7a3\s-]", "", text)
    # 공백이나 연속된 하이픈을 단일 하이픈으로 변환
    text = re.sub(r"[\s-]+", "-", text)
    # 앞뒤 하이픈 제거
    text = text.strip("-")
    return text.lower()
